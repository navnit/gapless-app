# macOS 0.1.0 Release Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.

**Goal:** Replace the all-platform release trigger with a signed, notarized, approval-gated macOS-only 0.1.0 release pipeline and a non-publishing manual candidate path.

**Architecture:** pubspec.yaml remains the sole application-version source, parsed by a small Dart utility that also supplies the SBOM version and rejects mismatched candidate or tag values. One macOS workflow handles manual candidates and v* tags, but only tag runs can enter the protected publication job; the existing all-platform tag workflow is removed so Windows and Linux remain verification-only targets.

**Tech Stack:** Flutter 3.44.4, Dart, GitHub Actions, Xcode command-line signing tools, Apple notarytool and stapler, hdiutil, GitHub Releases.

## Global Constraints

- The application version is exactly 0.1.0+1; the public tag is exactly v0.1.0.
- Release targets are exactly macos-arm64 on macos-14 and macos-x64 on macos-15-intel.
- Manual candidates sign, notarize, verify, and upload Actions artifacts without creating or modifying a GitHub Release.
- Tag publication requires both matrix jobs and the protected macos-release environment.
- Windows and Linux stay in .github/workflows/verify.yml but never appear in the tagged release matrix.
- The implementation does not create or push v0.1.0.
- Existing signing secret names and pinned action revisions remain unchanged.

Create tool/release/validate_release_version.dart:

~~~dart
import 'dart:io';

final class ReleaseVersion {
  const ReleaseVersion({required this.name, required this.buildNumber});

  final String name;
  final int buildNumber;

  static final _pubspecPattern = RegExp(
    r'^version:[ \t]*((?:0|[1-9][0-9]*)\.(?:0|[1-9][0-9]*)\.(?:0|[1-9][0-9]*))\+([1-9][0-9]*)[ \t]*\r?$',
    multiLine: true,
  );

  factory ReleaseVersion.parsePubspec(String source) {
    final matches = _pubspecPattern.allMatches(source).toList();
    if (matches.length != 1) {
      throw const FormatException(
        'pubspec.yaml must contain exactly one '
        'version: MAJOR.MINOR.PATCH+BUILD line.',
      );
    }
    return ReleaseVersion(
      name: matches.single.group(1)!,
      buildNumber: int.parse(matches.single.group(2)!),
    );
  }

  void requireReleaseName(String supplied) {
    if (supplied != name) {
      throw FormatException(
        'Release version $supplied does not match pubspec version $name.',
      );
    }
  }
}

Future<void> main(List<String> arguments) async {
  if (arguments.length != 2 || arguments.first != '--release-version') {
    stderr.writeln(
      'Usage: dart run tool/release/validate_release_version.dart '
      '--release-version MAJOR.MINOR.PATCH',
    );
    exitCode = 64;
    return;
  }
  try {
    final version = ReleaseVersion.parsePubspec(
      await File('pubspec.yaml').readAsString(),
    );
    version.requireReleaseName(arguments[1]);
    stdout.writeln(
      'Validated Gapless ${version.name}+${version.buildNumber}.',
    );
  } on FormatException catch (error) {
    stderr.writeln('error: ${error.message}');
    exitCode = 1;
  }
}
~~~

~~~yaml
on:
  workflow_dispatch:
    inputs:
      version:
        description: Gapless version from pubspec.yaml
        required: true
        default: '0.1.0'
        type: string
  push:
    tags: ['v*']

permissions:
  contents: read

jobs:
  build:
    strategy:
      fail-fast: false
      matrix:
        include:
          - target: macos-arm64
            os: macos-14
          - target: macos-x64
            os: macos-15-intel
    runs-on: ${{ matrix.os }}
    steps:
      - uses: actions/checkout@34e114876b0b11c390a56381ad16ebd13914f8d5
      - uses: subosito/flutter-action@1a449444c387b1966244ae4d4f8c696479add0b2
        with:
          flutter-version: 3.44.4
          cache: true
      - name: Resolve and validate release version
        id: version
        shell: bash
        run: |
          if [ "$GITHUB_EVENT_NAME" = workflow_dispatch ]; then
            release_version='${{ inputs.version }}'
          else
            release_version=${GITHUB_REF_NAME#v}
          fi
          dart run tool/release/validate_release_version.dart --release-version "$release_version"

          NOTARY_KEY_BASE64: ${{ secrets.MACOS_NOTARY_KEY_BASE64 }}
          NOTARY_KEY_ID: ${{ secrets.MACOS_NOTARY_KEY_ID }}
          NOTARY_ISSUER: ${{ secrets.MACOS_NOTARY_ISSUER }}

          test -n "$KEYCHAIN_PASSWORD" || { echo '::error::MACOS_KEYCHAIN_PASSWORD is required'; exit 1; }
          test -n "$NOTARY_KEY_BASE64" || { echo '::error::MACOS_NOTARY_KEY_BASE64 is required'; exit 1; }
          test -n "$NOTARY_KEY_ID" || { echo '::error::MACOS_NOTARY_KEY_ID is required'; exit 1; }
          test -n "$NOTARY_ISSUER" || { echo '::error::MACOS_NOTARY_ISSUER is required'; exit 1; }

          P12_PASSWORD: ${{ secrets.MACOS_P12_PASSWORD }}
          KEYCHAIN_PASSWORD: ${{ secrets.MACOS_KEYCHAIN_PASSWORD }}

            build/Gapless-${{ steps.version.outputs.value }}-${{ matrix.target }}.dmg

            build/THIRD_PARTY_NOTICES.md

        with:
          tag_name: ${{ github.ref_name }}
          name: Gapless ${{ github.ref_name }}
          generate_release_notes: true
          files: artifacts/*
~~~
